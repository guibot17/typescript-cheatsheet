Classes

class Person {
    lastName: string;
    firstName: string;
    constructor(firstName: string, lastName string) {
        this.firstName = firstName; // don't forget 'this.'
        this.lastName = lastName;
    };
    getFullName(): string { // no function keyword or assignment
        // the 'this.' is not optional
        return this.lastName + ", " + this.firstName;
    };
}
console.log(new Person("John", "Doe").getFullName()); // Doe, John
              
class User extends Person {
    email: string;
    login: string;
    constructor(firstName: string, lastName: string,
              email: string, login: string) {
        // have to call constructor on extended class!
        super(firstName, lastName);
        this.email = email;
        this.login = login;
    };
    toString(): string {
        return this.getFullName() + // call method of extended class
           " alias " + this.login + "(" + this.email + ")";
    };
}
