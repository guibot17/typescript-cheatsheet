// parameters can be optional
function greet(lastName: string, firstName?: string) { // note the '?'
    var greet: string = "hello ";
    if (firstName) greet += firstName + " ";
    return greet + lastName
}
greet("Doe"); // ok
greet("Doe", "Jane"); // ok
// optional parameters can have defaults
function greet(name: string, greetPhrase: string = "hello") {
    // [...]
}

console.log(greet("Jane", "hi")); // hi Jane

function join(delimiter: string, ... strings: string[]) {

Classes

class Person {
    lastName: string;
    firstName: string;
    constructor(firstName: string, lastName string) {
        this.firstName = firstName; // don't forget 'this.'
        this.lastName = lastName;
    };
    getFullName(): string { // no function keyword or assignment
        // the 'this.' is not optional
        return this.lastName + ", " + this.firstName;
    };
}
console.log(new Person("John", "Doe").getFullName()); // Doe, John
              
class User extends Person {
    email: string;
    login: string;
    constructor(firstName: string, lastName: string,
              email: string, login: string) {
        // have to call constructor on extended class!
        super(firstName, lastName);
        this.email = email;
        this.login = login;
    };
    toString(): string {
        return this.getFullName() + // call method of extended class
           " alias " + this.login + " (" + this.email + ")";
    };
}
class CodeLock {
    id: number; // implicitly public
    public manufacturer: string;
    private code: string;
    constructor(id: number, manufacturer: string, code: string) {

console.log(new CodeLock(1, "x", "top secret").id); // ok

class User {
    // properties, constructor, etc
    set role(role: PermissionRole) {
        if (role === Role.ADMIN && this.group != "admin") {
            throw new Error("Illegal setting of admin role");
        }
        this.role = role;
    }
}
new User("Jane Doe", "Users").role = PermissionRole.ADMIN; // boom

Interfaces

checkPermission(role: Role, hasRole: {role: Role}): bool {
    // this method takes every object
    // with a property role of type Role
    return true;
}
interface HasPermissions {
    roles: Role[];
    hasRole(role: Role): bool;
}
checkPermission(role: Role, user: HasPermissions): bool {
    return user && user.hasRole(role);
}

enum Size {S = "S", M = "M", XL};
